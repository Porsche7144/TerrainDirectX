// HLSL PixelShader

Texture2D shaderTexture;
SamplerState SampleType;

struct VS_OUTPUT
{
	float4 pos : SV_POSITION;
    float2 tex : TEXCOORD0;
};

float4 PS_PixelShader(VS_OUTPUT input) : SV_TARGET
{
	float4 textureColor;
	textureColor = shaderTexture.Sample(SampleType, input.tex);

	return textureColor;
}
